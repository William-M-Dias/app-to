from werkzeug.security import generate_password_hash, check_password_hash

class Usuario(db.Model):
    __tablename__ = 'usuarios'
    
    id = db.Column(db.Integer, primary_key=True)
    email = db.Column(db.String(120), unique=True, nullable=False)
    senha_hash = db.Column(db.String(255), nullable=False)
    nome = db.Column(db.String(100), nullable=False)

    # Função que embaralha a senha antes de salvar no banco
    def set_senha(self, senha):
        self.senha_hash = generate_password_hash(senha)

    # Função que checa se a senha digitada bate com a embaralhada
    def check_senha(self, senha):
        return check_password_hash(self.senha_hash, senha)

from flask import Blueprint, render_template, request, redirect, url_for, session, flash
from app.models.usuario import Usuario

auth_bp = Blueprint('auth_bp', __name__)

@auth_bp.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        email = request.form.get('email')
        senha = request.form.get('senha')
        
        # Procura o usuário no banco de dados
        usuario = Usuario.query.filter_by(email=email).first()

        if usuario and usuario.check_senha(senha):
            # Login com sucesso! Guarda a "chave" na sessão.
            session['usuario_id'] = usuario.id
            session['usuario_nome'] = usuario.nome
            return redirect(url_for('frontend_bp.index'))
        else:
            flash('E-mail ou senha incorretos. Tente novamente.', 'erro')

    return render_template('login.html')

@auth_bp.route('/logout')
def logout():
    session.clear() # Destrói a chave de acesso
    return redirect(url_for('auth_bp.login'))

# NOVA ROTA: O MOTOR DA AGENDA GLOBAL
# ==========================================
@consulta_bp.route('/agenda', methods=['GET'])
def agenda_global():
    # Importação local para evitar choque com outros módulos
    from app.models.paciente import Paciente 
    
    # Pega o dia de hoje, a partir da meia-noite
    hoje = datetime.now().replace(hour=0, minute=0, second=0, microsecond=0)
    
    # Filtra tudo que for 'Agendado' e que a data seja de hoje para a frente
    consultas = Consulta.query.filter(
        Consulta.status == 'Agendado',
        Consulta.data_hora >= hoje
    ).order_by(Consulta.data_hora.asc()).all()
    
    resultado = []
    for c in consultas:
        paciente = Paciente.query.get(c.paciente_id)
        if paciente:
            resultado.append({
                "id": c.id,
                "paciente_id": paciente.id,
                "nome_paciente": paciente.nome,
                "data": c.data_hora.strftime('%d/%m/%Y'),
                "hora": c.data_hora.strftime('%H:%M'),
                "tipo_sessao": c.tipo_sessao
            })
            
    return jsonify(resultado), 200

@consulta_bp.route('/paciente/<int:paciente_id>', methods=['GET'])
def listar_consultas_paciente(paciente_id):
    consultas = Consulta.query.filter_by(paciente_id=paciente_id).order_by(Consulta.data_hora.desc()).all()
    resultado = [{
        "id": c.id,
        "data_hora": c.data_hora.strftime('%d/%m/%Y %H:%M'),
        "tipo_sessao": c.tipo_sessao,
        "status": c.status,
        "evolucao_texto": c.evolucao_texto,
        "micro_metas": c.micro_metas or {}
    } for c in consultas]
    return jsonify(resultado), 200

@consulta_bp.route('/', methods=['POST'])
def registrar_evolucao():
    dados = request.get_json()
    
    if not dados or not dados.get('paciente_id') or not dados.get('data_hora'):
        return jsonify({"erro": "Dados incompletos para registo."}), 400
        
    try:
        data_string = dados['data_hora']
        if 'T' in data_string:
            data_string = data_string.replace('T', ' ')
        if len(data_string) == 10:
            data_string += ' 00:00'
            
        data_hora_consulta = datetime.strptime(data_string, '%Y-%m-%d %H:%M')
        
        nova_consulta = Consulta(
            paciente_id=dados['paciente_id'],
            data_hora=data_hora_consulta,
            tipo_sessao=dados.get('tipo_sessao', 'Intervenção de Rotina'),
            status=dados.get('status', 'Realizado'),
            evolucao_texto=dados.get('evolucao_texto', ''),
            micro_metas=dados.get('micro_metas', {})
        )
        
        db.session.add(nova_consulta)
        db.session.commit()
        return jsonify({"mensagem": "Registo salvo com sucesso!"}), 201
        
    except Exception as e:
        db.session.rollback()
        return jsonify({"erro": str(e)}), 500

@consulta_bp.route('/<int:id>', methods=['DELETE'])
def deletar_consulta(id):
    consulta = Consulta.query.get_or_404(id)
    try:
        db.session.delete(consulta)

        return jsonify({"mensagem": "Registro excluído com sucesso!"}), 200

    if not os.path.exists(upload_path):

                    if(this.form.micro_metas[dominio] === valor) {

                getGasColor(selecionado, botaoValor) {
                    if (selecionado !== botaoValor) return 'bg-white text-slate-400 border-slate-200 hover:bg-slate-50';
                    if (botaoValor === -2) return 'bg-rose-500 text-white border-rose-600 shadow-inner scale-105';
                    if (botaoValor === -1) return 'bg-orange-400 text-white border-orange-500 shadow-inner scale-105';
                    if (botaoValor === 0) return 'bg-slate-500 text-white border-slate-600 shadow-inner scale-105';
                    if (botaoValor === 1) return 'bg-clinica-400 text-white border-clinica-500 shadow-inner scale-105';

                    if (valor === 1) return 'Acima (Pouca ajuda)';
                    if (valor === 2) return 'Muito acima (Autónomo)';
                    return 'Não avaliado hoje';

                        payload.micro_metas.atividades = this.atividadesSelecionadas;
                        
                        // A MÁGICA DA VALIDAÇÃO: Remove tudo que for null, undefined ou array vazio
                        const metasLimpas = Object.fromEntries(
                            Object.entries(payload.micro_metas).filter(([chave, valor]) => {
                                if (valor === null || valor === undefined) return false;
                                if (Array.isArray(valor) && valor.length === 0) return false;
                                return true;
                            })
                        );

                        payload.micro_metas = {}; // Se não realizou, buraco negro vazio.

                    this.carregando = true;

                    ]).then(([dadosPacientes, dadosAgenda]) => {

                        this.agenda = Array.isArray(dadosAgenda) ? dadosAgenda : [];
                        const hoje = new Date().toLocaleDateString('pt-BR');
                        this.agendaHoje = this.agenda.filter(item => item.data === hoje);
                        this.carregando = false;
                    }).catch(err => {
                        console.error('Erro ao carregar dashboard:', err);
                        this.carregando = false;
                    });
                },

                getPacienteFoto(id) {
                    const p = this.pacientes.find(p => p.id === id);
                    return p ? p.foto_url : null;
                },

                    return dataStr === new Date().toLocaleDateString('pt-BR');
                },

                abrirModalCadastro() {
                    this.novoPaciente = { nome: '', data_nascimento: '', escolaridade: '', queixa_principal: '', nome_responsavel: '', telefone_responsavel: '', email_responsavel: '' };
                    this.diagnosticosSelecionados = [];
                    this.modalCadastroAberto = true;
                },

                    if (!this.novoPaciente.nome || !this.novoPaciente.data_nascimento) { alert("Nome e Nascimento obrigatórios."); return; }
                    this.salvando = true;
                    const payload = { ...this.novoPaciente, diagnostico: this.diagnosticosSelecionados.join(', ') };
                    fetch('/api/pacientes/', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    }).then(res => res.json()).then(data => {
                        if (data.id) { 
                            this.fecharModal(); 
                            this.inicializar(); 
                            if(this.modalAgendamentoAberto === false) {
                                this.novoAgendamento.paciente_id = data.id;
                                this.abrirModalAgendamento();
                            }
                        }
                        this.salvando = false;
                    }).catch(() => { this.salvando = false; });
                },
                
                abrirModalAgendamento() {
                    if (!this.novoAgendamento.data_hora) {
                        const agora = new Date();
                        agora.setMinutes(agora.getMinutes() - agora.getTimezoneOffset());
                        this.novoAgendamento.data_hora = agora.toISOString().slice(0, 16);
                    }
                    this.modalAgendamentoAberto = true;
                },
                fecharModalAgendamento() { this.modalAgendamentoAberto = false; },

                    if (!this.novoAgendamento.paciente_id || !this.novoAgendamento.data_hora) {

                        return;

                    }).catch(() => { this.salvandoAgendamento = false; });

                    return this.pacientes.filter(p => p.nome.toLowerCase().includes(t) || (p.diagnostico && p.diagnostico.toLowerCase().includes(t)));
                }

        function perfilPaciente(id) {
            return {
                pacienteId: id,
                paciente: null,
                avaliacoesPedi: [],
                ultimaAnamnese: null,
                historicoEvolucoes: [],
                carregando: true,
                subindoFoto: false, 
                
                modalEdicaoAberto: false,
                formEdicao: {},
                diagnosticosSelecionados: [],
                chartInstance: null,

                carregarDados() {
                    Promise.all([
                        fetch('/api/pacientes/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/pedi/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/anamnese/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/consultas/paciente/' + this.pacienteId).then(res => res.json())
                    ]).then(([dadosPaciente, dadosPedi, dadosAnamnese, dadosConsultas]) => {
                        this.paciente = dadosPaciente;
                        this.avaliacoesPedi = Array.isArray(dadosPedi) ? dadosPedi : [];
                        if (Array.isArray(dadosAnamnese) && dadosAnamnese.length > 0) this.ultimaAnamnese = dadosAnamnese[0];
                        this.historicoEvolucoes = Array.isArray(dadosConsultas) ? dadosConsultas : [];
                        this.carregando = false;
                        
                        // Após carregar os dados, renderiza o gráfico no próximo ciclo do DOM
                        this.$nextTick(() => {
                            this.renderizarGrafico();
                        });
                        
                    }).catch(err => {
                        console.error('Erro ao carregar prontuário:', err);
                        this.carregando = false;
                    });
                },

                // ADICIONADO: Verifica se existem sessões Realizadas com notas preenchidas
                temDadosParaGrafico() {
                    if (!this.historicoEvolucoes || this.historicoEvolucoes.length === 0) return false;
                    const validas = this.historicoEvolucoes.filter(e => e.status === 'Realizado' && e.micro_metas && 
                        (e.micro_metas.autocuidado !== null || e.micro_metas.mobilidade !== null || e.micro_metas.funcao_social !== null)
                    );
                    return validas.length > 0;
                },

                // ADICIONADO: Função que constrói e desenha o Gráfico Chart.js
                renderizarGrafico() {
                    if (!this.temDadosParaGrafico()) return;

                    const ctx = document.getElementById('gasChart');
                    if (!ctx) return;

                    // Filtra as realizadas e inverte para ficar cronológico (da mais antiga para a mais nova)
                    const sessoesValidas = this.historicoEvolucoes
                        .filter(e => e.status === 'Realizado' && e.micro_metas)
                        .slice()
                        .reverse();

                    // Formata as datas (Ex: "2023-10-25 14:00" -> "25/10")
                    const labels = sessoesValidas.map(e => {
                        let partes = e.data_hora.split(' ')[0].split('-');
                        if(partes.length === 3) return `${partes[2]}/${partes[1]}`;
                        return e.data_hora;
                    });

                    // Extrai as notas (se for null, não plota ponto naquele dia)
                    const dataAutocuidado = sessoesValidas.map(e => e.micro_metas.autocuidado !== undefined ? e.micro_metas.autocuidado : null);
                    const dataMobilidade = sessoesValidas.map(e => e.micro_metas.mobilidade !== undefined ? e.micro_metas.mobilidade : null);
                    const dataSocial = sessoesValidas.map(e => e.micro_metas.funcao_social !== undefined ? e.micro_metas.funcao_social : null);

                    if (this.chartInstance) {
                        this.chartInstance.destroy();
                    }

                    this.chartInstance = new Chart(ctx, {
                        type: 'line',
                        data: {
                            labels: labels,
                            datasets: [
                                { label: 'Autocuidado', data: dataAutocuidado, borderColor: '#0d9488', backgroundColor: '#0d9488', tension: 0.3, spanGaps: true, pointRadius: 4, pointHoverRadius: 6 },
                                { label: 'Mobilidade', data: dataMobilidade, borderColor: '#3b82f6', backgroundColor: '#3b82f6', tension: 0.3, spanGaps: true, pointRadius: 4, pointHoverRadius: 6 },
                                { label: 'Social', data: dataSocial, borderColor: '#f43f5e', backgroundColor: '#f43f5e', tension: 0.3, spanGaps: true, pointRadius: 4, pointHoverRadius: 6 }
                            ]
                        },
                        options: {
                            responsive: true,
                            maintainAspectRatio: false,
                            scales: {
                                y: {
                                    min: -2,
                                    max: 2,
                                    ticks: { stepSize: 1, font: { weight: 'bold' } },
                                    grid: {
                                        color: (context) => context.tick.value === 0 ? '#94a3b8' : '#f1f5f9',
                                        lineWidth: (context) => context.tick.value === 0 ? 2 : 1
                                    }
                                }
                            },
                            plugins: {
                                legend: { position: 'top', labels: { usePointStyle: true, boxWidth: 8, font: { family: 'Nunito', weight: 'bold' } } },
                                tooltip: { titleFont: { family: 'Nunito' }, bodyFont: { family: 'Nunito' } }
                            }
                        }
                    });
                },

                uploadFoto(event) {
                    const arquivo = event.target.files[0];
                    if (!arquivo) return;

                    this.subindoFoto = true;
                    const formData = new FormData();
                    formData.append('foto', arquivo);

                    fetch(`/api/pacientes/${this.pacienteId}/upload_foto`, {
                        method: 'POST',
                        body: formData
                    })
                    .then(res => res.json())
                    .then(data => {
                        if (data.url) {
                            this.paciente.foto_url = data.url;
                        } else {
                            alert("Erro ao processar imagem.");
                        }
                    })
                    .catch(err => alert("Erro na conexão com servidor."))
                    .finally(() => { this.subindoFoto = false; });
                },

                calcularIdade(dataNasc) {
                    if (!dataNasc) return 'Idade não informada';
                    const hoje = new Date();
                    const nascimento = new Date(dataNasc);
                    let idade = hoje.getFullYear() - nascimento.getFullYear();
                    const mes = hoje.getMonth() - nascimento.getMonth();
                    if (mes < 0 || (mes === 0 && hoje.getDate() < nascimento.getDate())) { idade--; }
                    return idade + (idade === 1 ? ' ano' : ' anos');
                },
                
                getEstiloGAS(valor) {
                    if (valor === -2) return 'bg-rose-50 text-rose-700 border-rose-200';
                    if (valor === -1) return 'bg-orange-50 text-orange-700 border-orange-200';
                    if (valor === 0) return 'bg-slate-100 text-slate-600 border-slate-200';
                    if (valor === 1) return 'bg-clinica-50 text-clinica-600 border-clinica-200';
                    if (valor === 2) return 'bg-clinica-100 text-clinica-800 border-clinica-300';
                    return 'bg-slate-50 text-slate-500 border-slate-100';
                },

                abrirEdicao() {
                    this.formEdicao = { ...this.paciente };
                    this.diagnosticosSelecionados = this.paciente.diagnostico ? this.paciente.diagnostico.split(', ') : [];
                    this.modalEdicaoAberto = true;
                },
                fecharEdicao() {
                    this.modalEdicaoAberto = false;
                },
                salvarEdicao() {
                    this.formEdicao.diagnostico = this.diagnosticosSelecionados.join(', ');
                    fetch('/api/pacientes/' + this.pacienteId, {
                        method: 'PUT',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(this.formEdicao)
                    }).then(res => {
                        if(res.ok) {
                            this.fecharEdicao();
                            this.carregando = true;
                            this.carregarDados();
                        } else { alert("Erro ao salvar."); }
                    });
                },
                apagarPaciente() {
                    if(confirm('ATENÇÃO: Tem certeza absoluta?')) {
                        fetch('/api/pacientes/' + this.pacienteId, { method: 'DELETE' })
                        .then(res => {
                            if(res.ok) window.location.href = '/';
                            else alert('Falha ao excluir.');
                        });
                    }
                },
                apagarEvolucao(idConsulta) {
                    if(confirm('Apagar esta evolução?')) {
                        fetch('/api/consultas/' + idConsulta, { method: 'DELETE' })
                        .then(res => {
                            if(res.ok) this.carregarDados();
                            else alert('Falha ao excluir.');
                        });
                    }
                }
            }
        }

        function avaliacaoPedi(id) {
            return {
                pacienteId: id,
                salvando: false,
                modo: 'rapida', // 'rapida' ou 'guiada'
                
                form: { 
                    score_autocuidado: 0, 
                    score_mobilidade: 0, 
                    score_funcao_social: 0, 
                    assistencia_autocuidado: '', 
                    assistencia_mobilidade: '', 
                    assistencia_social: '', 
                    observacoes: '' 
                },

                // Amostra estrutural real de perguntas do PEDI para a Dra. Edna. 
                // Ela pode adicionar quantas perguntas quiser aqui dentro do código futuramente.
                questoes: {
                    autocuidado: [
                        { texto: "Alimentação: Come alimentos com texturas variadas", resposta: null },
                        { texto: "Alimentação: Usa colher / garfo sem derramar muito", resposta: null },
                        { texto: "Higiene: Lava e seca as mãos de forma independente", resposta: null },
                        { texto: "Vestuário: Veste a camisa/camiseta sozinho", resposta: null },
                        { texto: "Vestuário: Consegue amarrar os sapatos ou fechar velcro", resposta: null }
                    ],
                    mobilidade: [
                        { texto: "Transferência: Entra e sai da cama sem ajuda", resposta: null },
                        { texto: "Transferência: Senta e levanta do vaso sanitário sozinho", resposta: null },
                        { texto: "Locomoção: Anda por toda a casa sem bater ou cair", resposta: null },
                        { texto: "Escadas: Sobe e desce um lance inteiro alternando os pés", resposta: null },
                        { texto: "Ambiente externo: Anda no quintal/rua sem perder o equilíbrio", resposta: null }
                    ],
                    social: [
                        { texto: "Compreensão: Entende frases complexas de duas ou três ordens", resposta: null },
                        { texto: "Expressão: Conta o que aconteceu no seu dia de forma clara", resposta: null },
                        { texto: "Resolução de Problemas: Pede ajuda quando algo dá errado", resposta: null },
                        { texto: "Brincadeira: Brinca de forma interativa e com regras com outras crianças", resposta: null },
                        { texto: "Segurança: Sabe evitar perigos claros (ex: faca, fogo, rua movimentada)", resposta: null }
                    ]
                },

                mudarModo(novoModo) {
                    this.modo = novoModo;
                    if(novoModo === 'guiada') {
                        // Força o recálculo imediato caso já tenha respostas gravadas
                        this.calcularEscore('autocuidado');
                        this.calcularEscore('mobilidade');
                        this.calcularEscore('social');
                    }
                },

                responder(categoria, index, valor) {
                    this.questoes[categoria][index].resposta = valor;
                    this.calcularEscore(categoria);
                },

                calcularEscore(categoria) {
                    if (this.modo !== 'guiada') return;

                    const arrayQuestoes = this.questoes[categoria];
                    // Quantas perguntas foram respondidas?
                    const respondidas = arrayQuestoes.filter(q => q.resposta !== null);
                    
                    if (respondidas.length === 0) {
                        this.form['score_' + categoria] = 0;
                        return;
                    }

                    // Soma dos "Capazes" (1)
                    const acertos = respondidas.reduce((soma, q) => soma + q.resposta, 0);
                    
                    // Converte para Escore Contínuo (Regra de 3 simples baseada no respondido)
                    // Na versão final clínica, pode-se aplicar a tabela de conversão exata do manual do PEDI
                    let percentual = Math.round((acertos / respondidas.length) * 100);
                    
                    this.form['score_' + categoria] = percentual;
                },

                salvarAvaliacao() {
                    // Validação básica do Módulo Rápido vs Guiado
                    if (this.form.score_autocuidado === 0 && this.form.score_mobilidade === 0 && this.form.score_funcao_social === 0) {
                        if(!confirm("Os escores estão zerados. Tem certeza que deseja salvar assim?")) return;
                    }

                    this.salvando = true;
                    fetch('/api/pedi/' + this.pacienteId, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(this.form)
                    }).then(res => {
                        if(res.ok) {
                            alert('Avaliação PEDI Salva com Sucesso!');
                            window.location.href = '/paciente/' + this.pacienteId;
                        } else {
                            throw new Error("Erro na gravação");
                        }
                    }).catch(err => { 
                        alert("Erro ao salvar PEDI.");
                        this.salvando = false; 
                    });
                }
            }
        }
    </script>

    <script>
        function relatorio(id) {
            return {
                pacienteId: id, 
                paciente: null, 
                avaliacoesPedi: [], 
                ultimaAnamnese: null, 
                historicoEvolucoes: [], 
                carregando: true,
                temDadosGrafico: false, // Controle de exibição do gráfico
                
                carregarDados() {
                    Promise.all([
                        fetch('/api/pacientes/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/pedi/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/anamnese/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/consultas/paciente/' + this.pacienteId).then(res => res.json())
                    ]).then(([dadosPaciente, dadosPedi, dadosAnamnese, dadosConsultas]) => {
                        this.paciente = dadosPaciente;
                        this.avaliacoesPedi = Array.isArray(dadosPedi) ? dadosPedi : [];
                        if (Array.isArray(dadosAnamnese) && dadosAnamnese.length > 0) this.ultimaAnamnese = dadosAnamnese[0];
                        this.historicoEvolucoes = Array.isArray(dadosConsultas) ? dadosConsultas.filter(c => c.status === 'Realizado') : [];
                        this.carregando = false;
                        
                        // Espera o Alpine atualizar o DOM e então renderiza o gráfico
                        this.$nextTick(() => {
                            this.renderizarGrafico();
                        });
                    }).catch(err => {
                        console.error('Erro ao carregar relatório:', err);
                        this.carregando = false;
                    });
                },
                
                calcularIdade(dataNasc) {
                    if (!dataNasc) return '';
                    const hoje = new Date(); const nascimento = new Date(dataNasc);
                    let idade = hoje.getFullYear() - nascimento.getFullYear();
                    const mes = hoje.getMonth() - nascimento.getMonth();
                    if (mes < 0 || (mes === 0 && hoje.getDate() < nascimento.getDate())) { idade--; }
                    return idade + (idade === 1 ? ' ano' : ' anos');
                },

                renderizarGrafico() {
                    // Filtra apenas as evoluções que têm dados GAS válidos
                    const evolucoesGAS = this.historicoEvolucoes.filter(e => 
                        e.micro_metas && 
                        (e.micro_metas.autocuidado !== null || e.micro_metas.mobilidade !== null || e.micro_metas.funcao_social !== null)
                    ).reverse(); // Inverte para ordem cronológica (da mais antiga para a mais nova)

                    if (evolucoesGAS.length === 0) return;
                    
                    this.temDadosGrafico = true;

                    // Extrai as datas (DD/MM) para o eixo X
                    const labels = evolucoesGAS.map(e => {
                        return e.data_hora.split(' ')[0].split('-').reverse().slice(0, 2).join('/');
                    });

                    // Calcula a média do GAS por sessão
                    const dataEvolucao = evolucoesGAS.map(e => {
                        let sum = 0;
                        let count = 0;
                        if (e.micro_metas.autocuidado !== null && e.micro_metas.autocuidado !== undefined) { sum += e.micro_metas.autocuidado; count++; }
                        if (e.micro_metas.mobilidade !== null && e.micro_metas.mobilidade !== undefined) { sum += e.micro_metas.mobilidade; count++; }
                        if (e.micro_metas.funcao_social !== null && e.micro_metas.funcao_social !== undefined) { sum += e.micro_metas.funcao_social; count++; }
                        return count > 0 ? parseFloat((sum / count).toFixed(1)) : 0;
                    });

                    // Garante que o canvas existe antes de plotar
                    this.$nextTick(() => {
                        const ctx = document.getElementById('gasChart');
                        if (!ctx) return;

                        new Chart(ctx, {
                            type: 'line',
                            data: {
                                labels: labels,
                                datasets: [{
                                    label: 'Média GAS',
                                    data: dataEvolucao,
                                    borderColor: '#0d9488', // clinica-600
                                    backgroundColor: 'rgba(13, 148, 136, 0.1)',
                                    borderWidth: 3,
                                    pointBackgroundColor: '#fff',
                                    pointBorderColor: '#0d9488',
                                    pointBorderWidth: 2,
                                    pointRadius: 4,
                                    fill: true,
                                    tension: 0.3
                                }]
                            },
                            options: {
                                responsive: true,
                                maintainAspectRatio: false,
                                scales: {
                                    y: {
                                        min: -2,
                                        max: 2,
                                        ticks: { stepSize: 1 }
                                    }
                                },
                                plugins: {
                                    legend: { display: false }
                                }
                            }
                        });
                    });
                }
            }
        }
    </script>
