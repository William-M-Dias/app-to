from werkzeug.security import generate_password_hash, check_password_hash

class Usuario(db.Model):
    __tablename__ = 'usuarios'
    
    id = db.Column(db.Integer, primary_key=True)
    email = db.Column(db.String(120), unique=True, nullable=False)
    senha_hash = db.Column(db.String(255), nullable=False)
    nome = db.Column(db.String(100), nullable=False)

    # Função que embaralha a senha antes de salvar no banco
    def set_senha(self, senha):
        self.senha_hash = generate_password_hash(senha)

    # Função que checa se a senha digitada bate com a embaralhada
    def check_senha(self, senha):
        return check_password_hash(self.senha_hash, senha)

from flask import Blueprint, render_template, request, redirect, url_for, session, flash
from app.models.usuario import Usuario

auth_bp = Blueprint('auth_bp', __name__)

@auth_bp.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        email = request.form.get('email')
        senha = request.form.get('senha')
        
        # Procura o usuário no banco de dados
        usuario = Usuario.query.filter_by(email=email).first()

        if usuario and usuario.check_senha(senha):
            # Login com sucesso! Guarda a "chave" na sessão.
            session['usuario_id'] = usuario.id
            session['usuario_nome'] = usuario.nome
            return redirect(url_for('frontend_bp.index'))
        else:
            flash('E-mail ou senha incorretos. Tente novamente.', 'erro')

    return render_template('login.html')

@auth_bp.route('/logout')
def logout():
    session.clear() # Destrói a chave de acesso
    return redirect(url_for('auth_bp.login'))

import uuid 

consulta_bp = Blueprint('consulta_bp', __name__, url_prefix='/api/consultas')

@consulta_bp.route('/agenda', methods=['GET'])
def agenda_global():
    from app.models.paciente import Paciente 
    
    hoje_brasil = datetime.utcnow() - timedelta(hours=3)
    hoje_meia_noite = hoje_brasil.replace(hour=0, minute=0, second=0, microsecond=0)
    
    consultas = Consulta.query.filter(
        Consulta.status == 'Agendado',
        Consulta.data_hora >= hoje_meia_noite
    ).order_by(Consulta.data_hora.asc()).all()
    
    resultado = []
    for c in consultas:
        paciente = Paciente.query.get(c.paciente_id)
        if paciente:
            resultado.append({
                "id": c.id,
                "paciente_id": paciente.id,
                "nome_paciente": paciente.nome,
                "data": c.data_hora.strftime('%d/%m/%Y'),
                "hora": c.data_hora.strftime('%H:%M'),
                "tipo_sessao": c.tipo_sessao
            })
            
    return jsonify(resultado), 200

@consulta_bp.route('/agenda_paginada', methods=['GET'])
def agenda_paginada():
    from app.models.paciente import Paciente 
    
    direcao = request.args.get('direcao', 'futuro')
    offset = int(request.args.get('offset', 0))
    limit = int(request.args.get('limit', 6))
    
    agora = datetime.utcnow() - timedelta(hours=3)
    linha_de_corte = agora - timedelta(minutes=60)
    
    if direcao == 'futuro':
        consultas = Consulta.query.filter(
            Consulta.data_hora >= linha_de_corte
        ).order_by(asc(Consulta.data_hora)).offset(offset).limit(limit).all()
    else:
        consultas = Consulta.query.filter(
            Consulta.data_hora < linha_de_corte
        ).order_by(desc(Consulta.data_hora)).offset(offset).limit(limit).all()
        
    resultado = []
    for c in consultas:
        paciente = Paciente.query.get(c.paciente_id)
        if paciente:
            data_obj = c.data_hora
            if isinstance(data_obj, str):
                try:
                    data_obj = datetime.strptime(data_obj, '%Y-%m-%d %H:%M:%S')
                except ValueError:
                    data_obj = datetime.strptime(data_obj, '%Y-%m-%d %H:%M')
                    
            resultado.append({
                "id": c.id,
                "paciente_id": paciente.id,
                "nome_paciente": paciente.nome,
                "data": data_obj.strftime('%d/%m/%Y'),
                "hora": data_obj.strftime('%H:%M'),
                "tipo_sessao": c.tipo_sessao,
                "status": c.status
            })
            
    return jsonify(resultado), 200

@consulta_bp.route('/', methods=['POST'])
def registrar_evolucao():
    dados = request.get_json()
    
    if not dados or not dados.get('paciente_id'):
        return jsonify({"erro": "Dados incompletos para registo."}), 400
        
    try:
        paciente_id = dados['paciente_id']
        status_recebido = dados.get('status', 'Agendado') 
        profissional_id = dados.get('profissional_id')
        duracao_minutos = int(dados.get('duracao_minutos', 50)) 
        
        data_string = dados.get('data_hora')
        if not data_string:
            data_string = (datetime.utcnow() - timedelta(hours=3)).strftime('%Y-%m-%d %H:%M')
            
        if 'T' in data_string:
            data_string = data_string.replace('T', ' ')
        if len(data_string) == 10:
            data_string += ' 00:00'
            
        data_hora_requisicao = datetime.strptime(data_string, '%Y-%m-%d %H:%M')
        data_fim_estimada = data_hora_requisicao + timedelta(minutes=duracao_minutos)
        data_apenas = data_hora_requisicao.date()
        
        consulta_existente = None
        
        if status_recebido in ['Realizado', 'Falta', 'Cancelado']:

            grupo_id = str(uuid.uuid4())

                        status='Agendado',

                        micro_metas={}
                    )
                    db.session.add(nova_consulta)

                data_atual_loop += timedelta(days=1)

@estatisticas_bp.route('/geral', methods=['GET'])
def obter_estatisticas():
    try:
        # 1. Total de Pacientes Ativos no Diretório
        total_pacientes = Paciente.query.count()

        # 2. Distribuição Clínica (Diagnósticos)
        # Como o diagnóstico pode vir separado por vírgula (ex: "TEA, TDAH"), 
        # puxamos tudo e contabilizamos via Python para garantir precisão
        todos_pacientes = Paciente.query.all()
        distribuicao_clinica = {}
        for p in todos_pacientes:
            if p.diagnostico:
                diags = [d.strip() for d in p.diagnostico.split(',')]
                for d in diags:
                    if d:
                        distribuicao_clinica[d] = distribuicao_clinica.get(d, 0) + 1
            else:
                distribuicao_clinica["Em Investigação"] = distribuicao_clinica.get("Em Investigação", 0) + 1

        # Prepara para o formato do gráfico do Chart.js
        grafico_diagnosticos = {
            "labels": list(distribuicao_clinica.keys()),
            "data": list(distribuicao_clinica.values())
        }

        # 3. Taxa de Assiduidade (Últimos 30 dias)
        hoje = datetime.utcnow() - timedelta(hours=3)
        trinta_dias_atras = hoje - timedelta(days=30)

        sessoes_periodo = Consulta.query.filter(
            Consulta.data_hora >= trinta_dias_atras,
            Consulta.data_hora <= hoje,
            Consulta.status.in_(['Realizado', 'Falta', 'Cancelado'])
        ).all()

        total_sessoes = len(sessoes_periodo)
        realizadas = sum(1 for s in sessoes_periodo if s.status == 'Realizado')
        faltas = sum(1 for s in sessoes_periodo if s.status in ['Falta', 'Cancelado'])

        # Evita divisão por zero
        taxa_assiduidade = 0
        if total_sessoes > 0:
            taxa_assiduidade = round((realizadas / total_sessoes) * 100, 1)

        # 4. Consultas da Semana (Previsão de Carga de Trabalho)
        inicio_semana = hoje - timedelta(days=hoje.weekday()) # Segunda-feira
        fim_semana = inicio_semana + timedelta(days=6, hours=23, minutes=59) # Domingo

        carga_semana = Consulta.query.filter(
            Consulta.data_hora >= inicio_semana,
            Consulta.data_hora <= fim_semana,
            Consulta.status != 'Cancelado'
        ).count()

        return jsonify({
            "total_pacientes": total_pacientes,
            "taxa_assiduidade": taxa_assiduidade,
            "total_sessoes_30d": total_sessoes,
            "sessoes_realizadas": realizadas,
            "sessoes_faltas": faltas,
            "carga_semana": carga_semana,
            "grafico_diagnosticos": grafico_diagnosticos
        }), 200

    except Exception as e:
        return jsonify({"erro": str(e)}), 500

@paciente_bp.route('/alertas', methods=['GET'])
def alertas_prontidao():
    try:
        pacientes = Paciente.query.all()
        alertas = []
        hoje = datetime.utcnow().date()

        for p in pacientes:
            ultimo_pedi = AvaliacaoPEDI.query.filter_by(paciente_id=p.id).order_by(desc(AvaliacaoPEDI.data_avaliacao)).first()
            ultimas_consultas = Consulta.query.filter_by(paciente_id=p.id, status='Realizado').order_by(desc(Consulta.data_hora)).limit(5).all()
            
            motivos = []
            
            if ultimo_pedi:
                data_pedi = ultimo_pedi.data_avaliacao
                if isinstance(data_pedi, str):
                    try:
                        data_pedi = datetime.strptime(data_pedi.split(' ')[0], '%Y-%m-%d').date()
                    except:
                        data_pedi = hoje
                elif isinstance(data_pedi, datetime):
                    data_pedi = data_pedi.date()
                
                dias_passados = (hoje - data_pedi).days
                meses_passados = dias_passados / 30
                if meses_passados >= 6:
                    motivos.append(f"A última avaliação PEDI foi há {int(meses_passados)} meses. Recomendada reavaliação de rotina.")
            else:
                if len(ultimas_consultas) > 0:
                    motivos.append("Paciente em intervenção, mas ainda não possui o PEDI base registado.")
                    
            gas_total = 0
            qtd_gas = 0
            for c in ultimas_consultas:
                if c.micro_metas and isinstance(c.micro_metas, dict):
                    valores = [v for k, v in c.micro_metas.items() if v is not None and isinstance(v, (int, float)) and k in ['autocuidado', 'mobilidade', 'funcao_social']]
                    if valores:
                        gas_total += sum(valores) / len(valores)
                        qtd_gas += 1
                        
            if qtd_gas >= 3:
                media_gas = gas_total / qtd_gas
                if media_gas >= 1.0:
                    motivos.append(f"Alta performance constante detectada (Média GAS: +{media_gas:.1f}). Criança pronta para subir de nível no PEDI.")
                    
            if motivos:
                alertas.append({
                    "paciente_id": p.id,
                    "nome_paciente": p.nome,
                    "foto_url": p.foto_url,
                    "motivos": motivos
                })
                
        return jsonify(alertas), 200
    except Exception as e:
        return jsonify({"erro": f"Erro ao processar alertas: {str(e)}"}), 500

        document.addEventListener('DOMContentLoaded', function() {
            var calendarEl = document.getElementById('calendar');
            var selectFiltro = document.getElementById('filtro-profissional');
            
            var isMobile = window.innerWidth < 768;
            
            var calendar = new FullCalendar.Calendar(calendarEl, {
                initialView: isMobile ? 'timeGridDay' : 'timeGridWeek',
                locale: 'pt-br',
                dayMinWidth: 120, // <--- MÁGICA DE VERDADE: Garante no mínimo 120px de largura por dia (cria scroll horizontal se faltar tela)
                headerToolbar: {
                    left: 'prev,next today',
                    center: 'title',
                    right: isMobile ? 'timeGridDay,timeGridWeek,listWeek' : 'timeGridWeek,timeGridDay,listWeek'
                },
                slotMinTime: '07:00:00',
                slotMaxTime: '20:00:00',
                allDaySlot: false,
                editable: true,
                selectable: true,
                eventDurationEditable: true, 
                
                events: function(info, successCallback, failureCallback) {
                    var profId = selectFiltro.value;
                    var url = '/api/consultas/calendario?start=' + info.startStr + '&end=' + info.endStr;
                    
                    if (profId) {
                        url += '&profissional_id=' + profId;
                    }

                    fetch(url)
                        .then(res => res.json())
                        .then(data => successCallback(data))
                        .catch(err => failureCallback(err));
                },
                
                eventClick: function(info) {
                    if (info.event.extendedProps.paciente_id) {
                        window.location.href = '/paciente/' + info.event.extendedProps.paciente_id;
                    }
                },
                
                eventDrop: function(info) {
                    if (!confirm(`Confirmar a mudança de horário de ${info.event.title}?`)) {
                        info.revert();
                        return;
                    }
                    fetch(`/api/consultas/${info.event.id}/mover`, {
                        method: 'PUT',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            nova_data: info.event.startStr,
                            nova_data_fim: info.event.endStr
                        })
                    }).then(res => {
                        if (!res.ok) { alert("Erro ao salvar mudança no banco."); info.revert(); }
                    }).catch(err => { alert("Falha de conexão."); info.revert(); });
                },
                
                eventResize: function(info) {
                     fetch(`/api/consultas/${info.event.id}/mover`, {
                        method: 'PUT',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            nova_data: info.event.startStr,
                            nova_data_fim: info.event.endStr
                        })
                    }).then(res => {
                        if (!res.ok) { alert("Erro ao salvar mudança de tempo."); info.revert(); }
                    });
                }
            });
            
            calendar.render();

            selectFiltro.addEventListener('change', function() {
                calendar.refetchEvents();
            });
        });
    </script>

                    if (botaoValor === 0) return 'bg-slate-500 text-white border-slate-600 shadow-inner scale-105';

                        // A MÁGICA DA VALIDAÇÃO: Remove tudo que for null, undefined ou array vazio

                                if (valor === null || valor === undefined) return false;
                                if (Array.isArray(valor) && valor.length === 0) return false;

        function gestaoDashboard() {
            return {
                pacientes: [],
                agendaHoje: [], 
                agendaVisual: [], 
                alertas: [], 
                
                profissionais: [
                    { id: 1, nome: 'Dra. Edna Nogueira (T.O.)' },
                    { id: 2, nome: 'Dr. Profissional 2' },
                    { id: 3, nome: 'Dra. Profissional 3' }
                ],
                
                carregando: true,
                pesquisa: '',
                mostrarLista: false, 
                modalCadastroAberto: false,
                salvando: false,
                novoPaciente: { nome: '', data_nascimento: '', escolaridade: '', queixa_principal: '', nome_responsavel: '', telefone_responsavel: '', email_responsavel: '' },
                diagnosticosSelecionados: [],
                listaDisfuncoes: ["TEA - Nível de Suporte 1", "TEA - Nível de Suporte 2", "TEA - Nível de Suporte 3", "TDAH", "TPS", "TDC", "DI - (Deficiência Intelectual)", "Paralisia Cerebral", "Síndrome de Down", "TOD", "Em Investigação", "Outro"],
                
                modalAgendamentoAberto: false,
                salvandoAgendamento: false,
                novoAgendamento: { paciente_id: '', data_hora: '', tipo_sessao: 'Intervenção', is_recorrente: false, meses_recorrencia: '1', profissional_id: '1' },
                
                offsetFuturo: 0,
                offsetPassado: 0,
                temMaisFuturo: true,
                temMaisPassado: true,

                inicializar() {
                    this.carregando = true;
                    Promise.all([
                        fetch('/api/pacientes/').then(res => res.json()),
                        fetch('/api/consultas/agenda').then(res => res.json()),
                        fetch('/api/pacientes/alertas').then(res => res.json())
                    ]).then(([dadosPacientes, dadosAgenda, dadosAlertas]) => {
                        this.pacientes = dadosPacientes;
                        this.alertas = Array.isArray(dadosAlertas) ? dadosAlertas : [];
                        
                        const hoje = new Date().toLocaleDateString('pt-BR');
                        this.agendaHoje = Array.isArray(dadosAgenda) ? dadosAgenda.filter(item => item.data === hoje) : [];
                        
                        this.carregarFuturo();
                        
                    }).catch(err => {
                        console.error('Erro ao carregar dashboard:', err);
                        this.carregando = false;
                    });
                },

                carregarFuturo() {
                    fetch(`/api/consultas/agenda_paginada?direcao=futuro&offset=${this.offsetFuturo}&limit=6`)
                    .then(res => res.json())
                    .then(data => {
                        if(data.length < 6) this.temMaisFuturo = false;
                        if (data.length > 0) {
                            this.agendaVisual = [...this.agendaVisual, ...data];
                            this.offsetFuturo += data.length;
                        }
                        this.carregando = false;
                    });
                },

                carregarPassado() {
                    fetch(`/api/consultas/agenda_paginada?direcao=passado&offset=${this.offsetPassado}&limit=6`)
                    .then(res => res.json())
                    .then(data => {
                        if (data.length < 6) this.temMaisPassado = false;
                        if (data.length > 0) {
                            this.agendaVisual = [...data.reverse(), ...this.agendaVisual]; 
                            this.offsetPassado += data.length;
                        }
                    });
                },

                getPacienteFoto(id) {
                    const p = this.pacientes.find(p => p.id === id);
                    return p ? p.foto_url : null;
                },

                isHoje(dataStr) {
                    return dataStr === new Date().toLocaleDateString('pt-BR');
                },

                abrirModalCadastro() {
                    this.novoPaciente = { nome: '', data_nascimento: '', escolaridade: '', queixa_principal: '', nome_responsavel: '', telefone_responsavel: '', email_responsavel: '' };
                    this.diagnosticosSelecionados = [];
                    this.modalCadastroAberto = true;
                },
                fecharModal() { this.modalCadastroAberto = false; },
                
                salvarPaciente() {
                    if (!this.novoPaciente.nome || !this.novoPaciente.data_nascimento) { alert("Nome e Nascimento obrigatórios."); return; }
                    this.salvando = true;
                    const payload = { ...this.novoPaciente, diagnostico: this.diagnosticosSelecionados.join(', ') };
                    fetch('/api/pacientes/', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    }).then(res => res.json()).then(data => {
                        if (data.id) { 
                            this.fecharModal(); 
                            this.agendaVisual = [];
                            this.offsetFuturo = 0;
                            this.offsetPassado = 0;
                            this.temMaisFuturo = true;
                            this.temMaisPassado = true;
                            this.inicializar(); 
                        }
                        this.salvando = false;
                    }).catch(() => { this.salvando = false; });
                },
                
                abrirModalAgendamento() {
                    if (!this.novoAgendamento.data_hora) {
                        const agora = new Date();
                        agora.setMinutes(agora.getMinutes() - agora.getTimezoneOffset());
                        this.novoAgendamento.data_hora = agora.toISOString().slice(0, 16);
                    }
                    this.modalAgendamentoAberto = true;
                },
                fecharModalAgendamento() { this.modalAgendamentoAberto = false; },

                salvarAgendamento() {
                    if (!this.novoAgendamento.paciente_id || !this.novoAgendamento.data_hora) {
                        alert("Por favor, selecione um paciente e o horário.");
                        return;
                    }
                    this.salvandoAgendamento = true;
                    
                    const payload = {
                        paciente_id: this.novoAgendamento.paciente_id,
                        profissional_id: parseInt(this.novoAgendamento.profissional_id),
                        data_hora: this.novoAgendamento.data_hora.replace('T', ' '),
                        status: 'Agendado',
                        tipo_sessao: this.novoAgendamento.tipo_sessao,
                        recorrente: this.novoAgendamento.is_recorrente,
                        meses_recorrencia: parseInt(this.novoAgendamento.meses_recorrencia),
                        evolucao_texto: '',
                        micro_metas: {}
                    };

                    fetch('/api/consultas/', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    }).then(res => res.json()).then(data => {
                        this.fecharModalAgendamento();
                        
                        this.agendaVisual = [];
                        this.offsetFuturo = 0;
                        this.offsetPassado = 0;
                        this.temMaisFuturo = true;
                        this.temMaisPassado = true;
                        this.inicializar(); 
                        
                        this.salvandoAgendamento = false;
                        this.novoAgendamento = { paciente_id: '', data_hora: '', tipo_sessao: 'Intervenção', is_recorrente: false, meses_recorrencia: '1', profissional_id: '1' };
                    }).catch(() => { this.salvandoAgendamento = false; });
                },

                pacientesFiltrados() {
                    if (this.pesquisa === '') return this.pacientes;
                    const t = this.pesquisa.toLowerCase();
                    return this.pacientes.filter(p => p.nome.toLowerCase().includes(t) || (p.diagnostico && p.diagnostico.toLowerCase().includes(t)));
                }
            }
        }

        function perfilPaciente(id) {
            return {
                pacienteId: id,
                paciente: null,
                avaliacoesPedi: [],
                ultimaAnamnese: null,
                historicoEvolucoes: [],
                carregando: true,
                subindoFoto: false, 
                
                modalEdicaoAberto: false,
                formEdicao: {},
                diagnosticosSelecionados: [],
                chartInstance: null,

                carregarDados() {
                    Promise.all([
                        fetch('/api/pacientes/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/pedi/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/anamnese/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/consultas/paciente/' + this.pacienteId).then(res => res.json())
                    ]).then(([dadosPaciente, dadosPedi, dadosAnamnese, dadosConsultas]) => {
                        this.paciente = dadosPaciente;
                        this.avaliacoesPedi = Array.isArray(dadosPedi) ? dadosPedi : [];
                        if (Array.isArray(dadosAnamnese) && dadosAnamnese.length > 0) this.ultimaAnamnese = dadosAnamnese[0];
                        this.historicoEvolucoes = Array.isArray(dadosConsultas) ? dadosConsultas : [];
                        this.carregando = false;
                        
                        this.$nextTick(() => {
                            this.renderizarGrafico();
                        });
                        
                    }).catch(err => {
                        console.error('Erro ao carregar prontuário:', err);
                        this.carregando = false;
                    });
                },

                // NOVO: Função para alterar o status do paciente (Ativo/Inativo)
                mudarStatus(novoStatus) {
                    if (this.paciente && this.paciente.status_clinico === novoStatus) return;

                    if (novoStatus === 'Inativo' && !confirm('Atenção: Mudar para "Pausa/Inativo" vai CANCELAR todos os agendamentos futuros deste paciente na agenda. Deseja continuar?')) {
                        return;
                    }

                    fetch(`/api/pacientes/${this.pacienteId}/status`, {
                        method: 'PUT',
                        headers: {'Content-Type': 'application/json'},
                        body: JSON.stringify({ status_clinico: novoStatus })
                    }).then(res => {
                        if (res.ok) {
                            this.paciente.status_clinico = novoStatus;
                            alert(`Status atualizado para ${novoStatus} com sucesso!`);
                            this.carregarDados(); // Recarrega para limpar as evoluções/agendamentos futuros deletados
                        } else {
                            alert('Erro ao atualizar status do paciente.');
                        }
                    }).catch(err => {
                        console.error(err);
                        alert('Erro na comunicação com o servidor.');
                    });
                },

                temDadosParaGrafico() {
                    if (!this.historicoEvolucoes || this.historicoEvolucoes.length === 0) return false;
                    const validas = this.historicoEvolucoes.filter(e => e.status === 'Realizado' && e.micro_metas && 
                        (e.micro_metas.autocuidado !== null || e.micro_metas.mobilidade !== null || e.micro_metas.funcao_social !== null)
                    );
                    return validas.length > 0;
                },

                renderizarGrafico() {
                    if (!this.temDadosParaGrafico()) return;

                    const ctx = document.getElementById('gasChart');
                    if (!ctx) return;

                    const sessoesValidas = this.historicoEvolucoes
                        .filter(e => e.status === 'Realizado' && e.micro_metas)
                        .slice()
                        .reverse();

                    const labels = sessoesValidas.map(e => {
                        let partes = e.data_hora.split(' ')[0].split('-');
                        if(partes.length === 3) return `${partes[2]}/${partes[1]}`;
                        return e.data_hora;
                    });

                    const dataAutocuidado = sessoesValidas.map(e => e.micro_metas.autocuidado !== undefined ? e.micro_metas.autocuidado : null);
                    const dataMobilidade = sessoesValidas.map(e => e.micro_metas.mobilidade !== undefined ? e.micro_metas.mobilidade : null);
                    const dataSocial = sessoesValidas.map(e => e.micro_metas.funcao_social !== undefined ? e.micro_metas.funcao_social : null);

                    if (this.chartInstance) {
                        this.chartInstance.destroy();
                    }

                    this.chartInstance = new Chart(ctx, {
                        type: 'line',
                        data: {
                            labels: labels,
                            datasets: [
                                { label: 'Autocuidado', data: dataAutocuidado, borderColor: '#0d9488', backgroundColor: '#0d9488', tension: 0.3, spanGaps: true, pointRadius: 4, pointHoverRadius: 6 },
                                { label: 'Mobilidade', data: dataMobilidade, borderColor: '#3b82f6', backgroundColor: '#3b82f6', tension: 0.3, spanGaps: true, pointRadius: 4, pointHoverRadius: 6 },
                                { label: 'Social', data: dataSocial, borderColor: '#f43f5e', backgroundColor: '#f43f5e', tension: 0.3, spanGaps: true, pointRadius: 4, pointHoverRadius: 6 }
                            ]
                        },
                        options: {
                            responsive: true,
                            maintainAspectRatio: false,
                            scales: {
                                y: {
                                    min: -2,
                                    max: 2,
                                    ticks: { stepSize: 1, font: { weight: 'bold' } },
                                    grid: {
                                        color: (context) => context.tick.value === 0 ? '#94a3b8' : '#f1f5f9',
                                        lineWidth: (context) => context.tick.value === 0 ? 2 : 1
                                    }
                                }
                            },
                            plugins: {
                                legend: { position: 'top', labels: { usePointStyle: true, boxWidth: 8, font: { family: 'Nunito', weight: 'bold' } } },
                                tooltip: { titleFont: { family: 'Nunito' }, bodyFont: { family: 'Nunito' } }
                            }
                        }
                    });
                },

                uploadFoto(event) {
                    const arquivo = event.target.files[0];
                    if (!arquivo) return;

                    this.subindoFoto = true;
                    const formData = new FormData();
                    formData.append('foto', arquivo);

                    fetch(`/api/pacientes/${this.pacienteId}/upload_foto`, {
                        method: 'POST',
                        body: formData
                    })
                    .then(res => res.json())
                    .then(data => {
                        if (data.url) {
                            this.paciente.foto_url = data.url;
                        } else {
                            alert("Erro ao processar imagem.");
                        }
                    })
                    .catch(err => alert("Erro na conexão com servidor."))
                    .finally(() => { this.subindoFoto = false; });
                },

                calcularIdade(dataNasc) {
                    if (!dataNasc) return 'Idade não informada';
                    const hoje = new Date();
                    const nascimento = new Date(dataNasc);
                    let idade = hoje.getFullYear() - nascimento.getFullYear();
                    const mes = hoje.getMonth() - nascimento.getMonth();
                    if (mes < 0 || (mes === 0 && hoje.getDate() < nascimento.getDate())) { idade--; }
                    return idade + (idade === 1 ? ' ano' : ' anos');
                },
                
                getEstiloGAS(valor) {
                    if (valor === -2) return 'bg-rose-50 text-rose-700 border-rose-200';
                    if (valor === -1) return 'bg-orange-50 text-orange-700 border-orange-200';
                    if (valor === 0) return 'bg-slate-100 text-slate-600 border-slate-200';
                    if (valor === 1) return 'bg-clinica-50 text-clinica-600 border-clinica-200';
                    if (valor === 2) return 'bg-clinica-100 text-clinica-800 border-clinica-300';
                    return 'bg-slate-50 text-slate-500 border-slate-100';
                },

                abrirEdicao() {
                    this.formEdicao = { ...this.paciente };
                    this.diagnosticosSelecionados = this.paciente.diagnostico ? this.paciente.diagnostico.split(', ') : [];
                    this.modalEdicaoAberto = true;
                },
                fecharEdicao() {
                    this.modalEdicaoAberto = false;
                },
                salvarEdicao() {
                    this.formEdicao.diagnostico = this.diagnosticosSelecionados.join(', ');
                    fetch('/api/pacientes/' + this.pacienteId, {
                        method: 'PUT',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(this.formEdicao)
                    }).then(res => {
                        if(res.ok) {
                            this.fecharEdicao();
                            this.carregando = true;
                            this.carregarDados();
                        } else { alert("Erro ao salvar."); }
                    });
                },
                apagarPaciente() {
                    if(confirm('ATENÇÃO: Tem certeza absoluta?')) {
                        fetch('/api/pacientes/' + this.pacienteId, { method: 'DELETE' })
                        .then(res => {
                            if(res.ok) window.location.href = '/';
                            else alert('Falha ao excluir.');
                        });
                    }
                },
                apagarEvolucao(idConsulta) {
                    if(confirm('Apagar esta evolução?')) {
                        fetch('/api/consultas/' + idConsulta, { method: 'DELETE' })
                        .then(res => {
                            if(res.ok) this.carregarDados();
                            else alert('Falha ao excluir.');
                        });
                    }
                }
            }
        }
    </script>

        function avaliacaoPedi(id) {
            return {
                pacienteId: id,
                salvando: false,
                modo: 'rapida', // 'rapida' ou 'guiada'
                
                form: { 
                    score_autocuidado: 0, 
                    score_mobilidade: 0, 
                    score_funcao_social: 0, 
                    assistencia_autocuidado: '', 
                    assistencia_mobilidade: '', 
                    assistencia_social: '', 
                    observacoes: '' 
                },

                // Amostra estrutural real de perguntas do PEDI para a Dra. Edna.

                questoes: {
                    autocuidado: [
                        { texto: "Alimentação: Come alimentos com texturas variadas", resposta: null },
                        { texto: "Alimentação: Usa colher / garfo sem derramar muito", resposta: null },
                        { texto: "Higiene: Lava e seca as mãos de forma independente", resposta: null },
                        { texto: "Vestuário: Veste a camisa/camiseta sozinho", resposta: null },
                        { texto: "Vestuário: Consegue amarrar os sapatos ou fechar velcro", resposta: null }
                    ],
                    mobilidade: [
                        { texto: "Transferência: Entra e sai da cama sem ajuda", resposta: null },
                        { texto: "Transferência: Senta e levanta do vaso sanitário sozinho", resposta: null },
                        { texto: "Locomoção: Anda por toda a casa sem bater ou cair", resposta: null },
                        { texto: "Escadas: Sobe e desce um lance inteiro alternando os pés", resposta: null },
                        { texto: "Ambiente externo: Anda no quintal/rua sem perder o equilíbrio", resposta: null }
                    ],
                    social: [
                        { texto: "Compreensão: Entende frases complexas de duas ou três ordens", resposta: null },
                        { texto: "Expressão: Conta o que aconteceu no seu dia de forma clara", resposta: null },
                        { texto: "Resolução de Problemas: Pede ajuda quando algo dá errado", resposta: null },
                        { texto: "Brincadeira: Brinca de forma interativa e com regras com outras crianças", resposta: null },
                        { texto: "Segurança: Sabe evitar perigos claros (ex: faca, fogo, rua movimentada)", resposta: null }
                    ]
                },

                mudarModo(novoModo) {
                    this.modo = novoModo;
                    if(novoModo === 'guiada') {
                        // Força o recálculo imediato caso já tenha respostas gravadas
                        this.calcularEscore('autocuidado');
                        this.calcularEscore('mobilidade');
                        this.calcularEscore('social');
                    }
                },

                responder(categoria, index, valor) {
                    this.questoes[categoria][index].resposta = valor;
                    this.calcularEscore(categoria);
                },

                calcularEscore(categoria) {
                    if (this.modo !== 'guiada') return;

                    const arrayQuestoes = this.questoes[categoria];
                    // Quantas perguntas foram respondidas?
                    const respondidas = arrayQuestoes.filter(q => q.resposta !== null);
                    
                    if (respondidas.length === 0) {
                        this.form['score_' + categoria] = 0;
                        return;
                    }

                    // Soma dos "Capazes" (1)
                    const acertos = respondidas.reduce((soma, q) => soma + q.resposta, 0);
                    
                    // Converte para Escore Contínuo (Regra de 3 simples baseada no respondido)
                    // Na versão final clínica, pode-se aplicar a tabela de conversão exata do manual do PEDI
                    let percentual = Math.round((acertos / respondidas.length) * 100);
                    
                    this.form['score_' + categoria] = percentual;
                },

                salvarAvaliacao() {
                    // Validação básica do Módulo Rápido vs Guiado
                    if (this.form.score_autocuidado === 0 && this.form.score_mobilidade === 0 && this.form.score_funcao_social === 0) {
                        if(!confirm("Os escores estão zerados. Tem certeza que deseja salvar assim?")) return;
                    }

                    this.salvando = true;
                    fetch('/api/pedi/' + this.pacienteId, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(this.form)
                    }).then(res => {
                        if(res.ok) {
                            alert('Avaliação PEDI Salva com Sucesso!');
                            window.location.href = '/paciente/' + this.pacienteId;
                        } else {
                            throw new Error("Erro na gravação");
                        }
                    }).catch(err => { 
                        alert("Erro ao salvar PEDI.");
                        this.salvando = false; 
                    });
                }
            }
        }

        function relatorio(id) {
            return {
                pacienteId: id,

                ultimaAnamnese: null, 
                historicoEvolucoes: [], 
                carregando: true,
                temDadosGrafico: false, 
                
                carregarDados() {
                    Promise.all([
                        fetch('/api/pacientes/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/pedi/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/anamnese/' + this.pacienteId).then(res => res.json()),
                        fetch('/api/consultas/paciente/' + this.pacienteId).then(res => res.json())
                    ]).then(([dadosPaciente, dadosPedi, dadosAnamnese, dadosConsultas]) => {
                        this.paciente = dadosPaciente;
                        this.avaliacoesPedi = Array.isArray(dadosPedi) ? dadosPedi : [];
                        if (Array.isArray(dadosAnamnese) && dadosAnamnese.length > 0) this.ultimaAnamnese = dadosAnamnese[0];
                        
                        // CORREÇÃO AQUI: adicionado o .reverse() para pegar as mais recentes!
                        this.historicoEvolucoes = Array.isArray(dadosConsultas) ? dadosConsultas.filter(c => c.status === 'Realizado').reverse() : [];
                        
                        this.carregando = false;
                        
                        this.$nextTick(() => {
                            this.renderizarGrafico();
                        });
                    }).catch(err => {
                        console.error('Erro ao carregar relatório:', err);
                        this.carregando = false;
                    });
                },
                
                calcularIdade(dataNasc) {
                    if (!dataNasc) return '';
                    const hoje = new Date(); const nascimento = new Date(dataNasc);
                    let idade = hoje.getFullYear() - nascimento.getFullYear();
                    const mes = hoje.getMonth() - nascimento.getMonth();
                    if (mes < 0 || (mes === 0 && hoje.getDate() < nascimento.getDate())) { idade--; }
                    return idade + (idade === 1 ? ' ano' : ' anos');
                },

                renderizarGrafico() {
                    // Aqui precisa de reverse() novamente porque o array original já foi invertido acima,
                    // e o Chart.js desenha o eixo X da esquerda (antigo) para a direita (recente).
                    const evolucoesGAS = this.historicoEvolucoes.filter(e => 
                        e.micro_metas && 
                        (e.micro_metas.autocuidado !== null || e.micro_metas.mobilidade !== null || e.micro_metas.funcao_social !== null)
                    ).slice().reverse(); 

                    if (evolucoesGAS.length === 0) return;
                    
                    this.temDadosGrafico = true;

                    const labels = evolucoesGAS.map(e => {
                        return e.data_hora.split(' ')[0].split('-').reverse().slice(0, 2).join('/');
                    });

                    const dataEvolucao = evolucoesGAS.map(e => {
                        let sum = 0;
                        let count = 0;
                        if (e.micro_metas.autocuidado !== null && e.micro_metas.autocuidado !== undefined) { sum += e.micro_metas.autocuidado; count++; }
                        if (e.micro_metas.mobilidade !== null && e.micro_metas.mobilidade !== undefined) { sum += e.micro_metas.mobilidade; count++; }
                        if (e.micro_metas.funcao_social !== null && e.micro_metas.funcao_social !== undefined) { sum += e.micro_metas.funcao_social; count++; }
                        return count > 0 ? parseFloat((sum / count).toFixed(1)) : 0;
                    });

                    this.$nextTick(() => {
                        const ctx = document.getElementById('gasChart');
                        if (!ctx) return;

                        new Chart(ctx, {
                            type: 'line',
                            data: {
                                labels: labels,
                                datasets: [{
                                    label: 'Média GAS',
                                    data: dataEvolucao,
                                    borderColor: '#0d9488', 
                                    backgroundColor: 'rgba(13, 148, 136, 0.1)',
                                    borderWidth: 3,
                                    pointBackgroundColor: '#fff',
                                    pointBorderColor: '#0d9488',
                                    pointBorderWidth: 2,
                                    pointRadius: 4,
                                    fill: true,
                                    tension: 0.3
                                }]
                            },
                            options: {
                                responsive: true,
                                maintainAspectRatio: false,
                                scales: {
                                    y: {
                                        min: -2,
                                        max: 2,
                                        ticks: { stepSize: 1 }
                                    }
                                },
                                plugins: {
                                    legend: { display: false }
                                }
                            }
                        });
                    });
                }
            }
        }
